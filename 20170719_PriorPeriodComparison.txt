Temp_Data:
LOAD
	 Floor(rand()*1000, 0.01)						as Amount
	,Date(Today()-RowNo(), 'DD MMM YYYY')			as Date
	,'Dimension1'									as Dimension1
AutoGenerate(3000)
;

Data:
LOAD 
	*
	,Year(Date)			as Year
	,Month(Date)		as Month
	,Day(Date)			as Day
	,Date(MonthStart(Date), 'MM-YYYY')		as MonthYear
Resident Temp_Data
;

DROP Table Temp_Data
