///$tab Test Data
//generate Test Data from today - 24 months earlier
LET tempEndDate = num(Today());
LET tempStartDate = num(AddMonths($(tempEndDate),-12)) -1;

DateMaster:
LOAD
		*
		,month(Date)	as Month
		,year(Date)		as Year
		,num(rand() * 1000, '#0') as Sales
		,date(MonthStart(Date),'MMM-YYYY') as YearMonth
;
LOAD
	date($(tempStartDate) + IterNo(),'MMM-DD-YYYY')	as Date
AutoGenerate 1
While $(tempStartDate) + IterNo() <= $(tempEndDate)
;

SET tempStartDate=;
SET tempEndDate=;
